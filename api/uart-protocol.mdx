## Overview

The `UARTProtocol` class implements a custom serial protocol for bidirectional communication between the RC controller, TX module, and RX module. It handles channel data, telemetry, status updates, and commands.

## Class Definition

Location: `lib/UARTProtocol/UARTProtocol.h`

```cpp
class UARTProtocol {
public:
    UARTProtocol(HardwareSerial* serial);
    
    void begin(uint32_t baud = UART_PROTOCOL_BAUDRATE);
    
    // Send methods
    bool sendChannels(const uint16_t channels[8]);
    bool sendTelemetry(const TelemetryData* data);
    bool sendStatus(const StatusData* data);
    bool sendCommand(UARTMsgType cmd);
    bool sendAck(UARTMsgType ackedCmd);
    bool sendError(uint8_t errorCode);
    bool sendPing();
    bool sendPong();
    
    void loop();
    
    // Callback setters
    void setOnChannels(OnChannelsCallback cb);
    void setOnTelemetry(OnTelemetryCallback cb);
    void setOnStatus(OnStatusCallback cb);
    void setOnCommand(OnCommandCallback cb);
    void setOnAck(OnAckCallback cb);
    void setOnError(OnErrorCallback cb);
    void setOnPong(OnPongCallback cb);
    
    // Statistics
    uint32_t getPacketsSent() const;
    uint32_t getPacketsReceived() const;
    uint32_t getPacketsDropped() const;
    void resetStats();
};
```

## Message Types

```cpp
enum UARTMsgType : uint8_t {
    UART_MSG_CHANNELS = 0x01,      // RC channel data
    UART_MSG_CMD_PAIR = 0x10,      // Enter pairing mode
    UART_MSG_CMD_BOND = 0x11,      // Check bond status
    UART_MSG_CMD_RESTART = 0x12,   // Restart device
    UART_MSG_CMD_STATUS_REQ = 0x13,// Request status
    UART_MSG_PING = 0x14,          // Device availability check
    UART_MSG_PONG = 0x15,          // Device availability response
    UART_MSG_TELEMETRY = 0x20,     // Telemetry data
    UART_MSG_STATUS = 0x21,        // Status data
    UART_MSG_ACK = 0x22,           // Command acknowledgment
    UART_MSG_ERROR = 0x23          // Error response
};
```

```cpp
void sendTelemetryToRC() {
    if (millis() - lastTelemetrySent < TELEMETRY_INTERVAL) return;
    lastTelemetrySent = millis();
    
    TelemetryData telem;
    telem.rssi = last_rssi;
    telem.snr = last_snr;
    
    if (rx_battery_valid && (millis() - lastBatteryUpdate) < 5000) {
        telem.rxBattMv = rx_battery_mv;
        telem.rxBattPct = rx_battery_pct;
    } else {
        telem.rxBattMv = 0;
        telem.rxBattPct = 0;
    }
    
    // Calculate link quality (0-100%) based on RSSI
    if (last_rssi > -50) {
        telem.linkQuality = 100;
    } else if (last_rssi < -120) {
        telem.linkQuality = 0;
    } else {
        telem.linkQuality = (uint8_t)map(last_rssi, -120, -50, 0, 100);
    }
    
    uartProto.sendTelemetry(&telem);
}
```

## Frame Format

```
[SYNC][LEN][TYPE][PAYLOAD...][CRC]
```

- **SYNC:** 0xA5 (1 byte)
- **LEN:** Payload length (1 byte)
- **TYPE:** Message type (1 byte)
- **PAYLOAD:** Message data (0-60 bytes)
- **CRC:** CRC8 checksum (1 byte)

## Complete Usage Example

From `src/rc_crsf_main.cpp:277`:

```cpp
UARTProtocol uartProto(&Serial2);

void setup() {
    // Initialize UART protocol
    Serial2.setTX(CRSF_TX_PIN);
    Serial2.setRX(CRSF_RX_PIN);
    uartProto.begin(UART_PROTOCOL_BAUDRATE);
    
    // Set callbacks
    uartProto.setOnTelemetry(onTelemetryReceived);
    uartProto.setOnStatus(onStatusReceived);
    uartProto.setOnAck(onAckReceived);
    uartProto.setOnError(onErrorReceived);
    uartProto.setOnPong(onPongReceived);
}

void loop() {
    // Send channels at 250Hz
    static unsigned long lastChannelSend = 0;
    if (millis() - lastChannelSend >= 4) {
        lastChannelSend = millis();
        uartProto.sendChannels(channels);
    }
    
    // Process incoming messages
    uartProto.loop();
}
```
